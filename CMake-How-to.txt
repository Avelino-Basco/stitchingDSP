(1) Create Folder
(2) Ctrl + Shift + P to bring up Command Path
Alt: Click command path then type ">"
(3) Type ">Cmake Quick Start"
(4) Select amd_x86_x64
(5) Type ">Cmake Quick Start" again and enter a name
(6) C++
(7) Executable
(8)
CMakeLists.txt must have
"
cmake_minimum_required(VERSION 3.0.0)
project(needle VERSION 0.1.0 LANGUAGES C CXX)

include(CTest)
enable_testing()

find_package( OpenCV REQUIRED )
include_directories( ${OpenCV_INCLUDE_DIRS} )

add_executable(needle main.cpp)

target_link_libraries( needle ${OpenCV_LIBS} )

set(CPACK_PROJECT_NAME ${PROJECT_NAME})
set(CPACK_PROJECT_VERSION ${PROJECT_VERSION})
include(CPack)
"
